The default debug formatter for TupleStruct types.

§Example

use bevy_reflect::Reflect;
#[derive(Reflect)]
struct MyTupleStruct(usize);

let my_tuple_struct: &dyn Reflect = &MyTupleStruct(123);
println!("{:#?}", my_tuple_struct);

// Output:

// MyTupleStruct (
//   123,
// )
